Recommended Supabase RLS policies

Apply these policies in the Supabase Dashboard for the lessons table and the lessons storage bucket.

-- Lessons table policies create policy "Admins can manage lessons" on public.lessons for all to authenticated using ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.email = 'your_admin_email@example.com' ) ) with check ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.email = 'your_admin_email@example.com' ) ); create policy "Authenticated users can read lessons" on public.lessons for select to authenticated using (true); -- Storage bucket policies create policy "Admins can upload lesson files" on storage.objects for insert to authenticated with check ( bucket_id = 'lessons' and exists ( select 1 from public.profiles p where p.id = auth.uid() and p.email = 'your_admin_email@example.com' ) ); create policy "Admins can update and delete lesson files" on storage.objects for update, delete to authenticated using ( bucket_id = 'lessons' and exists ( select 1 from public.profiles p where p.id = auth.uid() and p.email = 'your_admin_email@example.com' ) ); create policy "Authenticated users can view lesson files" on storage.objects for select to authenticated using (bucket_id = 'lessons');
